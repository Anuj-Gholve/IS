    <script>
        function power_mod(base, exp, mod) {
            let result = 1;
            base = base % mod;
            while (exp > 0) {
                if (exp % 2 === 1)
                    result = (result * base) % mod;
                exp = Math.floor(exp / 2);
                base = (base * base) % mod;
            }
            return result;
        }

        function exchange() {
            let P = parseInt(document.getElementById("prime").value);
            let G = parseInt(document.getElementById("generator").value);
            let a = parseInt(document.getElementById("alicePrivate").value);
            let b = parseInt(document.getElementById("bobPrivate").value);

            // Step 1: Compute public keys
            let alicePublic = power_mod(G, a, P);   // A = G^a mod P
            let bobPublic = power_mod(G, b, P);     // B = G^b mod P

            // Step 2: Compute shared secrets
            let aliceSecret = power_mod(bobPublic, a, P);   // s = B^a mod P
            let bobSecret = power_mod(alicePublic, b, P);   // s = A^b mod P

            let html = `<div class="result">
<b>Public Parameters:</b> P = ${P}, G = ${G}

<b>Alice's private key (a):</b> ${a}
<b>Bob's private key (b):</b> ${b}

<b>Step 1 - Public Keys:</b>
  Alice sends: A = G^a mod P = ${G}^${a} mod ${P} = ${alicePublic}
  Bob sends:   B = G^b mod P = ${G}^${b} mod ${P} = ${bobPublic}

<b>Step 2 - Shared Secret:</b>
  Alice computes: B^a mod P = ${bobPublic}^${a} mod ${P} = ${aliceSecret}
  Bob computes:   A^b mod P = ${alicePublic}^${b} mod ${P} = ${bobSecret}
</div>`;

            if (aliceSecret === bobSecret) {
                html += `<p class="match">Shared Secret Key = ${aliceSecret} (Both match!)</p>`;
            } else {
                html += `<p style="color:red;">Keys do NOT match!</p>`;
            }

            document.getElementById("output").innerHTML = html;
        }
    </script>
